 17         /**
 18          * 메뉴 아이디를  기준으로 신규 윈도우 화면을 생성하고 open 시킴
 19          * @param {XaComp} this(현재폼)
 20          * @param {object} arguments
 21          * @return
 22          * @example
 23          *  var oObj = {
 24          *       ds : this.ds_menu,
 25          *       nRow : e.row,
 26          *       aArg : []
 27          *   };
 28          *        Iject.Mdi.call(this,oObj);
 29          * @memberOf Iject.Mdi
 30          */
 31         call : function(XComp,oObj)
 32         {
 33 
 34             if(!Eco.isEmpty(oObj) && typeof(oObj) !=  "object") return;
 35             var gdsOpen = application.gds_openMenu;    //열린 dataset
 36             var gdsMenu = application.gds_menu;       //global dataset
 37             var ds   = oObj.ds;    //넘어온 dataset
 38             var nRow = oObj.nRow;  //선택된 현재 row
 39             var aArgs = Eco.isEmpty(oObj.oArg) ? "" : oObj.oArg ;   //넘어온 arguments
 40             var sMenuId = ds.getColumn(nRow,  Iject.$.MENU_ID);
 41 
 42             var winid = gdsOpen.lookup(Iject.$.MENU_ID,sMenuId,Iject.$.WIN_ID);
 43 
 44             if (!Eco.isEmpty(winid))
 45             {
 46 
 47                 if (Iject.$.mdiFrame.form.isActiveFrame(winid, aArgs) == true)
 48                 {
 49                     Iject.$.mdiFrame.form.fn_moveTab(winid);
 50                     return;
 51                 }
 52             }
 53 
 54             Iject.Mdi.newMdi(sMenuId, nRow, aArgs);
 55 
 56         },
 57 
 58 
 59 
 60         /**
 61          * 신규 생성된 윈도우(프레임) 화면을 gds_openMenu에 등록
 62          * @private
 63          * @param {string} winid: 윈도우 생성 키
 64          * @param {string} menuid: 메뉴아이디
 65          * @param {string}  strTitle:타이틀
 66          * @return {boolen} true/false
 67          * @example
 68          *     Iject.Mdi.setOpenMenuDs(winid, sMenuid, strTitle);   // 열린메뉴 화면 삽입
 69          * @memberOf Iject.Mdi
 70          */
 71         setOpenMenuDs : function(winid, menuid, strTitle)
 72         {
 73             var gdsOpen = application.gds_openMenu;  //열린 dataset
 74             var curRow = gdsOpen.addRow();
 75                 gdsOpen.setColumn(curRow, "WIN_ID", winid);
 76                 gdsOpen.setColumn(curRow, "MENU_ID", menuid);
 77                 gdsOpen.setColumn(curRow, "TITLE", strTitle);
 78         },
 79 
 80         /**
 81          *  넘어온 아규먼트 처리 설정
 82          * @param  {XaComp} this(현재form)
 83          * @return
 84          * @example
 85          * Iject.Mdi.setOwnFrameArgu(this);
 86          * @memberOf Iject.Mdi
 87          */
 88         setOwnFrameArgu : function(o)
 89         {
 90             Iject.$.w_winkey  = o.getOwnerFrame().arguments["winKey"];
 91             Iject.$.w_pageurl = o.getOwnerFrame().arguments["pageUrl"];
 92             Iject.$.w_menunm  = o.getOwnerFrame().arguments["menuNm"];
 93             Iject.$.w_mneuid  = o.getOwnerFrame().arguments["menuId"];
 94             Iject.$.w_title   = Iject.$.w_menunm+" ("+ Iject.$.w_mneuid+")";
 95 
 96         },
 97 
 98         /**
 99          * menu open 실행되는 event
100          * @private
101          * @param {XaComp} this(현재폼)
102          * @param {object} 펼침접힘실행버튼
103          * @return
104          * @example
105          *   Iject.Mdi.leftMenuAction(this,obj);
106          * @memberOf Iject.Mdi
107          */
108         leftMenuAction : function(o,oBtn)
109         {
110 
111 
112             if(oBtn.cssclass =="btn_TF_LeftMenuOpen")
113             {
114                 Iject.$.hFrame.set_separatesize("230,*");
115                 oBtn.set_cssclass("btn_TF_LeftMenuClose");
116             }
117             else
118             {
119                 Iject.$.hFrame.set_separatesize("0,*");
120                 oBtn.set_cssclass("btn_TF_LeftMenuOpen");
121             }
122 
123 
124             if (nexacro.Browser != "Runtime" && nexacro.BrowserVersion > 9)
125             {
126 
127                 var objLeft = Iject.$.leftFrame.form.div_left;
128                 objLeft.set_left(0);
129                 objLeft.set_left(-50);
130                 $(Iject.Html5.jquery(objLeft)).animate({left:0},'fast');
131 
132             }
133         },

136         /**
137          * gds_openMenu의 해당 Row의 정보를 기준으로 신규 윈도우 화면을 생성하고 open 시킴
138          * @param {string} menuId
139          * @param {number} gds_openMenu의rowpostion
140          * @param {array} arguments
141          * @return 
142          * @example
143          *  Iject.Mdi.newMdi(sMenuId, nRow, aArgs);
144          * @memberOf Iject.Mdi
145          */
146         newMdi : function(sMenuid,  nRow, aArgs)
147         {
148 
149             var gdsOpen = application.gds_openMenu;  //열린 dataset
150             var gdsMenu = application.gds_menu;      //global dataset
151             var winid = "win" + sMenuid + "_" + gdsOpen.getRowCount() + "_" + parseInt(Math.random() * 1000);
152             //var winid = "win" + sMenuid;
153             var strTitle = gdsMenu.lookup(Iject.$.MENU_ID,sMenuid,Iject.$.MENU_NM);
154             var sPageUrl = gdsMenu.lookup(Iject.$.MENU_ID, sMenuid,Iject.$.PAGE_URL);
155             var sMenuNm  =  gdsMenu.lookup(Iject.$.MENU_ID, sMenuid, Iject.$.MENU_NM);
156 
157             if(Eco.isEmpty(sPageUrl)) return;   //pageURl 이 없으면 return
158 
159             Iject.Mdi.setOpenMenuDs(winid, sMenuid, strTitle);   // 열린메뉴 화면 삽입
160 
161             var objNewWin = new ChildFrame;
162                 objNewWin.init(winid, "absolute", 0, 0, Iject.$.workFrame.getOffsetWidth() - 0, Iject.$.workFrame.getOffsetHeight() - 0);
163 
164                 Iject.$.workFrame.addChild(winid, objNewWin);
165 
166                 objNewWin.set_tooltiptext(winid);
167                 objNewWin.arguments = [];
168                 objNewWin.set_dragmovetype("all");
169                 objNewWin.set_showtitlebar(false);
170                 objNewWin.set_resizable(true);
171                 objNewWin.set_openstatus("maximize");
172                 objNewWin.set_titletext( strTitle);
173                 objNewWin.arguments["winKey"]   =  winid;
174                 objNewWin.arguments["menuId"]   =  sMenuid;
175                 objNewWin.arguments["menuNm"]   =  sMenuNm;
176                 objNewWin.arguments["pageUrl"]  =  sPageUrl;
177                 objNewWin.arguments["aArgs"]    =  aArgs;
178                 objNewWin.set_formurl("Frame::WorkFrame.xfdl");
179                 objNewWin.show();
180 
181             Iject.$.mdiFrame.form.fn_addTab(winid, strTitle);   //mdi tab button add
182             Iject.$.mdiFrame.form.isActiveFrame(winid);
183 
184         }
185
