  5 if (!JsNamespace.exist("Iject.Runtime")) {
  6 
  7     /**
  8      * @namespace
  9      * @name Iject.Runtime
 10      * @author
 11      * @memberof! Iject.Runtime
 12      */
 13     JsNamespace.declare("Iject.Runtime", {
 14 
 15         /**
 16          * login 처리
 17          * @param {XaComp} this(현재Form)
 18          * @return
 19          * @example
 20          *
 21          * @memberOf Iject.Runtime
 22          */
 23         loginR : function(o)
 24         {
 25             //application.mainframe.set_visible("false");
 26             var pThis = o.mainframe;
 27             var nScreenWidth;
 28             var nScreenHeight;
 29             var arrScreenSize;
 30             var nMoniterIndex = system.getMonitorIndex(system.cursorx, system.cursory); //모니터번호
 31             var objScreenSize = system.getScreenRect(nMoniterIndex);
 32             var nScreenLeft = objScreenSize.left;
 33             var nScreenTop = objScreenSize.top;
 34             var nScreenRight = objScreenSize.right;
 35             var nScreenBottom = objScreenSize.bottom;
 36                 nScreenWidth = nScreenRight - nScreenLeft;
 37                 nScreenHeight = nScreenBottom - nScreenTop;
 38 
 39             Iject.$.vFrameSet.set_separatesize("*,0,0");
 40             application.mainframe.set_showtitlebar(false);
 41             application.mainframe.set_left(nScreenLeft+(nScreenWidth/2)-180);
 42             application.mainframe.set_top(nScreenTop+(nScreenHeight/2)-180);
 43             application.mainframe.set_width(Iject.$.loginSize.width);
 44             application.mainframe.set_height(Iject.$.loginSize.height);
 45 
 46             var oData ={
 47                 sId            : "login",             //callback id
 48                 sUrl           : "Frame::login.xfdl",   //url
 49                 bshowtitlebar  : false,     //title bar
 50                 bAutoSize      : true,       //autosize
 51                 bResize        : false,       //resize
 52                 bShowstatusbar : false,     //statusbar
 53                 sOpenalign     : "center middle",  //align
 54                 sBackground    : "transparnet",
 55                 oArgs : {}                   // argument
 56             };
 57 
 58             //팝업 테스트
 59             Iject.showModal(pThis,oData,function(svcid,variant){
 60 
 61                 /*
 62                  * popup calllback
 63                  *@param {object}this.fobj
 64                  *@param {string} this.svcid
 65                  *@param {string} this.variant
 66                  */
 67                 var pThis= this.fobj;
 68                 Iject.Runtime.setFrame(pThis);   //frame size open
 69 
 70             });
 71 
 72         },
 73         
 74         /**
 75          * Runtime logout
 76          * @param
 77          * @return
 78          * @example
 79          *
 80          * @memberOf Iject.Runtime
 81          */
 82         logoutR : function ()
 83         {
 84             if(Iject.confirm("로그아웃 하시겠습니까?")) application.exit();
 85         },
 86 
 87         /**
 88          * login 처리가 완료됬을때 frame size 조정
 89          * @param {XaComp} this(현재Form)
 90          * @return
 91          * @example
 92          *  Iject.setFrame(this);
 93          * @memberOf Iject.Runtime
 94          */
 95        setFrame: function(o)
 96        {
 97            application.mainframe.set_visible(false);
 98            Iject.setSeprateFrame(); //로그인 처리후 메인 프레임 값 setting
 99 
100            application.mainframe.set_left(0);
101            application.mainframe.set_top(0);
102            application.mainframe.set_width(Iject.$.mainframeSize.width);
103            application.mainframe.set_height(Iject.$.mainframeSize.height);
104            application.mainframe.set_showtitlebar(true);
105            application.mainframe.set_visible(true);
106 
107        }
108 
109 
110     });
